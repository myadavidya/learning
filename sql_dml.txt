JOINING:


SELECT studentname,code
FROM SCHOOL1 s1,SCHOOL2 s2
WHERE s1.subject=s2.subject;



MAXIMUM:

SELECT MAX(marks)
FROM SCHOOL2


MINIMUM:

SELECT MIN(marks)
FROM SCHOOL2


LIKE:

SELECT teachername
FROM SCHOOL1
WHERE teachername LIKE '%A%';

NOT LIKE:

SELECT studentname
FROM SCHOOL2
WHERE studentname NOT LIKE '%A%';


COUNT:


SELECT COUNT(code)
FROM SCHOOL1;


AVERAGE:

SELECT AVG(marks)
FROM SCHOOL2;


IN:

SELECT code
FROM SCHOOL1
WHERE code IN (1001,1002);

NOT IN:


SELECT code
FROM SCHOOL1
WHERE code NOT IN (1001,1002);


BETWEEN:

SELECT *
FROM SCHOOL1 
WHERE experience BETWEEN 5 AND 10;

NOT BETWEEN:

SELECT *
FROM SCHOOL1 
WHERE experience  NOT BETWEEN 5 AND 10;


DESCENDING:


SELECT code
FROM SCHOOL1 
ORDER BY code DESC;


ASCENDING ORDER:

SELECT code
FROM SCHOOL1 
ORDER BY code ;

SUM:

SELECT sum(marks)
FROM SCHOOL2;

GROUP BY:

SELECT COUNT(experience),periods
FROM SCHOOL1
